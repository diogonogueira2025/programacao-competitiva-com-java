# Saída
---

Para a saída podemos usar o System.out.println(), System.out.print ou o System.out.printf():
```java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.IOException;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader in = new BufferedReader(
            new InputStreamReader(System.in)
        );
        
        // entrada: "123 456 monkey"
        String[] tokens = in.readLine().trim().split(" ");

        int a = Integer.parseInt(tokens[0]);  // 123
        int b = Integer.parseInt(tokens[1]);  // 456
        String x = tokens[2];                 // "monkey"

        System.out.println(a + " " + b);  // imprime e quebra linha
        System.out.print(x);              // imprime sem quebrar linha
        System.out.printf("%d %d%n", a, b); // formatação estilo C
        
        in.close();
    }
}
```
